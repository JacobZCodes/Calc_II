$$\begin{aligned} e^x &\approx f(x) = c_0 + c_1 x + c_2 x^2 + c_3 x^3 + \dots + c_n x^n \\ f(0) &= e^0 \\ \implies c_0 &= 1\end{aligned}$$


$$\begin{aligned} f'(x) &= c_1 + 2c_2 x + 3c_3 x^2 + \dots + n c_n x^{n-1} && \text{and} &\quad \frac{d}{dx}(e^x) = e^x \\ f'(0) &= c_1 && \text{and} &\quad e^0 = 1 \\ \implies c_1 &= 1\end{aligned}$$


$$\begin{aligned} f''(x) &= 2c_2 + 6c_3 x + \dots + n(n-1) c_n x^{n-2} && \text{and} &\quad \frac{d^2}{dx^2}(e^x) = e^x \\ f''(0) &= 2c_2 && \text{and} &\quad e^0 = 1 \\ \implies c_2 &= \frac{1}{2}\end{aligned}$$


$$\begin{aligned} f'''(x) &= 6c_3 + \dots + n(n-1)(n-2)c_n x^{n-3} && \text{and} &\quad \frac{d^3}{dx^3}(e^x) = e^x \\ f'''(0) &= 6c_3 && \text{and} &\quad e^0 = 1 \\ \implies c_3 &= \frac{1}{6}\end{aligned}$$


$$\begin{aligned} f^{(n)}(x) &= n! c_n && \text{and} &\quad \frac{d^n}{dx^n}(e^x) = e^x \\ f^{(n)}(0) &= n! && \text{and} &\quad e^0 = 1 \\ \implies c_n &= \frac{1}{n!}\end{aligned}$$


$$\begin{aligned} f(x) = 1 + x + \frac{1}{2}x^2 + \frac{1}{6}x^3 + \dots + \frac{1}{n!}x^n\end{aligned}$$


$$\boxed{f(x) = \frac{x^0}{0!} + \frac{x^1}{1!} + \frac{x^2}{2!} + \frac{x^3}{3!} + \dots + \frac{x^n}{n!}}$$
